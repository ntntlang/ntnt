Quick Example

fn withdraw(account: BankAccount, amount: Int) -> Bool
    requires amount > 0
    ensures result == true implies account.balance >= 0
{
    if account.balance >= amount {
        account.balance = account.balance - amount
        return true
    }
    return false
}
