Explore these examples to learn NTNT:

Hello World

// hello.tnt
let greeting = "Hello, World!"
print(greeting)

fn greet(name) {
    return "Hello, " + name + "!"
}

print(greet("NTNT"))

Pattern Matching

fn describe(opt) {
    match opt {
        Some(v) => print("Got: " + v),
        None => print("Nothing")
    }
}

let result = Some(42)
describe(result)  // "Got: 42"

HTTP API

import { json, text } from "std/http/server"

fn get_status(req) {
    return json(map {
        "status": "ok",
        "version": "0.1.6"
    })
}

fn get_user(req) {
    let id = req.params.id
    return json(map { "id": id })
}

get("/status", get_status)
get("/users/{id}", get_user)
listen(3000)
