Quick Start Code

Hello World

// hello.tnt
let greeting = "Hello, World!"
print(greeting)

fn greet(name) {
    return "Hello, " + name + "!"
}

print(greet("NTNT"))

Contracts

fn divide(a: Int, b: Int) -> Int
    requires b != 0  // Precondition
    ensures result * b == a  // Postcondition
{
    return a / b
}

HTTP Server

import { json } from "std/http/server"

fn get_status(req) {
    return json(map {
        "status": "ok",
        "version": "0.1.6"
    })
}

get("/api/status", get_status)
listen(3000)

Running Examples

To run any example:

ntnt run examples/hello.tnt
ntnt run examples/http_server.tnt
ntnt test examples/http_server.tnt --get /api/status
